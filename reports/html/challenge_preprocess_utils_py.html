31def is_high_season(fecha): 

32 """ 

33 Based on Scheduled date and time of the flight. 

34  

35 1 if `Date-I` is between  

36 Dec-15 and Mar-3,  

37 or Jul-15 and Jul-31,

39 0 otherwise. 

40 """

42 fecha = datetime.strptime(fecha, '%Y-%m-%d %H:%M:%S') 

43 range1_min = datetime.strptime('15-Dec', '%d-%b').replace(year = fecha_anio) 

44 range1_max = datetime.strptime('31-Dec', '%d-%b').replace(year = fecha_anio) 

45 range2_min = datetime.strptime('1-Jan', '%d-%b').replace(year = fecha_anio)
